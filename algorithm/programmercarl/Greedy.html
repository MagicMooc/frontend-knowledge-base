var largestSumAfterKNegations = function (nums, k) {
  let count = 0;
  nums.forEach((item) => {
    if (item < 0) {
      count++;
    }
  });

  if (k <= count) {
    for (let i = 0; i < k; i++) {
      nums[i] = -nums[i];
    }
  } else {
    for (let i = 0; i < count; i++) {
      nums[i] = -nums[i];
    }
    nums.sort((a, b) => a - b);
    let t = k - count;
    while (t--) {
      nums[0] = -nums[0];
    }
  }

    return total + cur;
  }, 0);
  return res;
};
